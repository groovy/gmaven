155   /**
156    * Log a properties object.
157    */
158   private void logProperties(final Properties source) {
159     Map<String, String> map = Maps.fromProperties(source);
160     List<String> keys = Lists.newArrayList(map.keySet());
161     Collections.sort(keys);
162     for (String key : keys) {
163       log("  {}={}", key, map.get(key));
164     }
165   }

167   /**
168    * Load a resources as a {@link Properties}.
169    */
170   private Properties loadResourceProperties(final String resourceName) throws IOException {
171     Properties props = new Properties();
172 
173     URL resource = getClass().getResource(resourceName);
174     if (resource != null) {
175       log("Loading properties from: {}", resource);
176       InputStream stream = resource.openStream();
177       try {
178         props.load(stream);
179       }
180       finally {
181         stream.close();
182       }
183     }
184 
185     return props;
186   }

188   /**
189    * Try to figure out what the version of the plugin we are testing.  Falls back to {@link #DEFAULT_UNDER_TEST_VERSION}.
190    */
191   private String detectUnderTestVersion() throws IOException {
192     Properties props = loadResourceProperties("/META-INF/maven/org.codehaus.gmaven/groovy-maven-plugin/pom.properties");
193     String version = props.getProperty("version");
194 
195     // this can happen in IDE context, complain and move on
196     if (version == null) {
197       version = DEFAULT_UNDER_TEST_VERSION;
198       log("WARNING: unable to detect under-test version!");
199     }
200 
201     return version;
202   }

204   /**

207   private File createSettingsFile(final File buildRepository) throws Exception {
208     return interpolate(

216   /**
217    * Create a new file interpolating from given template.
218    */
219   private File interpolate(final File template, final Map<String, String> context) throws Exception {
220     String content = FileUtils.readFileToString(template);
221     Interpolator interpolator = new StringSearchInterpolator();
222     interpolator.addValueSource(new MapBasedValueSource(context));
223     content = interpolator.interpolate(content);
224     File file = util.createTempFile(template.getName());
225     FileUtils.write(file, content);
226     return file;
227   }

229   /**
230    * Detect the repository used to build the plugin under test.
231    */
232   private File detectBuildRepository() {
233     String path = System.getProperty("buildRepository");
234     if (path != null) {
235       return new File(path).getAbsoluteFile();
236     }
237 
238     // fall back to default
239     File userHome = new File(System.getProperty("user.home"));
240     return new File(userHome, ".m2/repository");
241   }

250   /**
251    * Create a verifier for the given project-name, cloning its contents to test work directory.
252    */
253   protected MavenVerifier verifier(final String projectName) throws Exception {
254     File sourceDir = util.resolveFile("src/test/it-projects/" + projectName);
255     File projectDir = testdir();
256 
257     log("Copying {} -> {}", sourceDir, projectDir);
258     FileUtils.copyDirectory(sourceDir, projectDir);
259 
260     return new MavenVerifier(projectDir, settingsFile)
261         .addArg("-V")
262         .setProperty("underTest.version", underTestVersion)
263         .setProperty("groovy.version", groovyVersion)
264         .setProperty("gmaven.logging", "DEBUG");
265   }
