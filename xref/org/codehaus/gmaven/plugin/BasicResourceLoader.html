13  package org.codehaus.gmaven.plugin;
14  
15  import java.net.MalformedURLException;
16  import java.net.URL;
17  
18  import javax.annotation.Nullable;
19  
20  import org.codehaus.gmaven.adapter.ResourceLoader;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import static com.google.common.base.Preconditions.checkNotNull;
25  
26  /**
27   * Basic {@link ResourceLoader} implementation.
28   *
29   * @since 2.0
30   */
31  public class BasicResourceLoader
32      implements ResourceLoader
33  {
34    protected static final String DOT_GROOVY = ".groovy";
35  
36    protected final Logger log = LoggerFactory.getLogger(getClass());
37  
38    protected final ClassLoader classLoader;
39  
40    public BasicResourceLoader(final ClassLoader classLoader) {
41      this.classLoader = checkNotNull(classLoader);
42    }
43  
44    @Nullable
45    public URL loadResource(final String name) throws MalformedURLException {
46      return resolve(name, classLoader);
47    }
48  
49    @Nullable
50    protected URL resolve(final String className, final ClassLoader classLoader) throws MalformedURLException {
51      log.trace("Resolve; class-name: {}, class-loader: {}", className, classLoader);
52  
53      String name = resourceName(className);
54  
55      log.trace("Resolve; resource: {}", name);
56  
57      boolean tcl = false;
58      URL resource = classLoader.getResource(name);
59  
60      // try from tcl if given CL does not resolve
61      if (resource == null) {
62        resource = Thread.currentThread().getContextClassLoader().getResource(name);
63        tcl = true;
64      }
65  
66      log.trace("Resource: {}, tcl: {}", resource, tcl);
67  
68      return resource;
69    }
70  
71    /**
72     * Translate class-name into resource-name.
73     */
74    protected String resourceName(final String className) {
75      String name = className;
76  
77      if (!name.startsWith("/")) {
78        name = "/" + name;
79      }
80  
81      if (!name.endsWith(DOT_GROOVY)) {
82        name = name.replace('.', '/');
83        name += DOT_GROOVY;
84      }
85  
86      return name;
87    }
88  }
