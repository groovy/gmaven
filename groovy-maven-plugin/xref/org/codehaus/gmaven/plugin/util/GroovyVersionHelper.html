13  package org.codehaus.gmaven.plugin.util;
14  
15  import java.lang.reflect.Method;
16  
17  import javax.annotation.Nullable;
18  
19  import com.google.common.annotations.VisibleForTesting;
20  import org.codehaus.plexus.component.annotations.Component;
21  import org.codehaus.plexus.component.annotations.Requirement;
22  import org.eclipse.aether.version.Version;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import static com.google.common.base.Preconditions.checkNotNull;
27  
28  /**
29   * Groovy version helper.
30   *
31   * @since 2.0
32   */
33  @Component(role=GroovyVersionHelper.class)
34  public class GroovyVersionHelper
35  {
36    private final Logger log = LoggerFactory.getLogger(getClass());
37  
38    @Requirement
39    private VersionHelper versionHelper;
40  
41    public GroovyVersionHelper() {}
42  
43    @VisibleForTesting
44    public GroovyVersionHelper(final VersionHelper versionHelper) {
45      this.versionHelper = checkNotNull(versionHelper);
46    }
47  
48    @Nullable
49    public Version detectVersion(final ClassLoader classLoader) {
50      checkNotNull(classLoader);
51  
52      log.trace("Detecting Groovy version; class-loader: {}", classLoader);
53  
54      // Modern versions of Groovy expose the version via GroovySystem.getVersion()
55      String raw = getVersion(classLoader, "groovy.lang.GroovySystem", "getVersion");
56      if (raw == null) {
57        // Older versions of Groovy expose the version via InvokerHelper.getVersion()
58        raw = getVersion(classLoader, "org.codehaus.groovy.runtime.InvokerHelper", "getVersion");
59      }
60  
61      log.trace("Raw version: {}", raw);
62      if (raw == null) {
63        return null;
64      }
65  
66      Version version = versionHelper.parseVersion(raw);
67      log.trace("Version: {}", version);
68      return version;
69    }
70  
71    /**
72     * Get version from Groovy version helper utilities via reflection.
73     */
74    private String getVersion(final ClassLoader classLoader, final String className, final String methodName) {
75      log.trace("Getting version; class-loader: {}, class-name: {}, method-name: {}",
76          classLoader, className, methodName);
77  
78      try {
79        Class<?> type = classLoader.loadClass(className);
80        Method method = type.getMethod(methodName);
81        Object result = method.invoke(null);
82        if (result != null) {
83          return result.toString();
84        }
85      }
86      catch (Throwable e) {
87        log.trace("Unable determine version from: {}", className);
88      }
89  
90      return null;
91    }
92  }
