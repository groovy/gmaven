26  @Mojo(name = "resolve")
27  public class ResolveMojo
28      extends MojoSupport
29  {
30      private static final String DEFAULT_GROOVY_VERSION = "2.0.1";
31  
32      @Parameter(property="groovy.version", defaultValue=DEFAULT_GROOVY_VERSION)
33      private String groovyVersion;
34  
35      @Component
36      private MavenProject project;
37  
38      @Component
39      private RepositorySystem repositorySystem;
40  
41      @Parameter(property="repositorySystemSession") // @Component does not work here
42      private RepositorySystemSession repositorySession;
43  
44      @Override
45      protected void doExecute() throws Exception {
46          String artifactId = "org.codehaus.groovy:groovy:" + groovyVersion;
47  
48          CollectRequest collectRequest = new CollectRequest();
49          collectRequest.setRepositories(project.getRemotePluginRepositories());
50          collectRequest.setRoot(new Dependency(new DefaultArtifact(artifactId), JavaScopes.RUNTIME));
51  
52          List<ArtifactResult> artifacts = repositorySystem.resolveDependencies(repositorySession, collectRequest, null);
53          // FIXME: Filter out optional dependencies
54  
55          ClassWorld world = new ClassWorld();
56          ClassLoader parentCl = Thread.currentThread().getContextClassLoader();
57          String realmId = ClassRealms.uniqueId();
58  
59          ClassRealm realm = world.newRealm(realmId, parentCl);
60  
61          // TODO: Sort out if we want to expose any Maven bits at all, if not, could likely make the parentCl an empty realm
62          //ClassRealms.setStrategy(realm, new ParentFirstStrategy(realm));
63  
64          log.info("Class-path:");
65          for (ArtifactResult artifact : artifacts) {
66              log.info("  {}", artifact);
67              URL url = artifact.getArtifact().getFile().toURI().toURL();
68              log.info("    {}", url);
69              realm.addURL(url);
70          }
71  
72          VersionDetector versionDetector = new VersionDetector();
73          String tmp = versionDetector.detectVersion(realm);
74          if (tmp == null) {
75              log.error("Unable to detect Groovy version");
76              return;
77          }
78          Version version = versionDetector.parseVersion(tmp);
79          log.debug("Version: {}", version);
80      }
