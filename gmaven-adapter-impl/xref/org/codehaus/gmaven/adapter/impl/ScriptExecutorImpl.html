63          classSource, classLoader, resourceLoader, context);

65      GroovyClassLoader gcl = runtime.createGroovyClassLoader(classLoader, resourceLoader);
66  
67      CompilerConfiguration cc = new CompilerConfiguration();
68  
69      // TODO: How, if at all, do we want to expose compiler configuration customizations?
70  
71      Binding binding = runtime.createBinding(context);
72      GroovyShell shell = new GroovyShell(gcl, binding, cc);
73  
74      if (options != null) {
75        configureOptions(shell, options);
76      }
77  
78      GroovyCodeSource codeSource = runtime.createGroovyCodeSource(classSource);
79      try {
80        return shell.evaluate(codeSource);
81      }
82      finally {
83        gcl.clearCache();
84      }
